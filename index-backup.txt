      window.tomtomSDKLoading = new Promise((resolve, reject) => {
        // try local vendor path first
        const localCss = '/vendor/tomtom/maps/maps-web.min.css';
        const localJs = '/vendor/tomtom/maps/maps-web.min.js';

        function loadJs(src){
          const s = document.createElement('script');
          s.src = src;
          s.crossOrigin = 'anonymous';
          s.onload = resolve;
          s.onerror = () => reject(new Error('TomTom SDK failed to load: ' + src));
          document.head.appendChild(s);
        }

        // always append CSS (no onload event)
        const link = document.createElement('link');
        link.rel = 'stylesheet';
        link.href = localCss;
        link.crossOrigin = 'anonymous';
        link.onerror = () => { /* ignore */ };
        document.head.appendChild(link);

        // attempt JS - use fetch with crossorigin to avoid ORB issues
        fetch(localJs, {method:'HEAD', mode: 'no-cors'}).then(resp => {
          if (resp.ok) loadJs(localJs);
          else {
            // fallback CDN with crossorigin attribute
            loadJs('https://api.tomtom.com/maps-sdk-for-web/cdn/6.x/6.22.0/maps/maps-web.min.js');
          }
        }).catch(() => {
          // fallback CDN with crossorigin attribute
          loadJs('https://api.tomtom.com/maps-sdk-for-web/cdn/6.x/6.22.0/maps/maps-web.min.js');
        });
      });
      return window.tomtomSDKLoading;

      console.log('index.html restore check:', {pendingSection, pendingScrollMain, pendingScrollWindow, pendingScrollLegacy, pendingScroll});
      if(pendingSection){
        // mark that we're in restore mode so other callers don't override
        window.__restoringSection = true;
        console.log('Restoring section:', pendingSection);
        // Directly perform the load for restore to avoid queuing the request
        window.__performLoadSection(pendingSection).then(()=>{
          // One deterministic scroll after the section has rendered.
          // Wait for layout stabilization (two animation frames) and retry a few times
          console.log('Section loaded, scrolling to:', pendingScroll);
          // Decide whether to restore main-content scroll or window/document scroll
          const preferMain = Boolean(pendingScrollMain && pendingScrollMain > 0);
          (async function waitAndScroll(target, useMain){
            const main = document.getElementById('main-content');
            function doScroll(){
              try{
                if(useMain && main && typeof main.scrollTo === 'function'){
                  main.scrollTo(0, target);
                } else {
                  // Try document.scrollingElement first for broader compatibility
                  const se = document.scrollingElement || document.documentElement || document.body;
                  if(se && typeof se.scrollTo === 'function') se.scrollTo(0, target);
                  else window.scrollTo(0, target);
                }
              }catch(e){ console.error('scroll error', e); }
            }
            // allow initial layout to settle (two frames)
            await new Promise(r=>requestAnimationFrame(()=>requestAnimationFrame(r)));
            doScroll();
            // Wait for images inside main-content to load (or timeout) then try again
            try{
              const imgs = Array.from((main||document).querySelectorAll('img'));
              const pending = imgs.filter(img => !img.complete);
              if(pending.length){
                // Wait for all current images to load or 2500ms timeout
                await Promise.race([
                  Promise.all(pending.map(i => new Promise(res => { i.addEventListener('load', res); i.addEventListener('error', res); }))),
                  new Promise(res => setTimeout(res, 2500))
                ]);
              }

              // Also observe the DOM briefly for any images added after render (e.g., scripts inserting images)
              const addedImgs = [];
              const mo = new MutationObserver(muts => {
                for(const m of muts){
                  for(const node of Array.from(m.addedNodes || [])){
                    if(node && node.querySelectorAll){
                      Array.from(node.querySelectorAll('img')).forEach(i => addedImgs.push(i));
                    }
                    if(node && node.tagName === 'IMG') addedImgs.push(node);
                  }
                }
              });
              mo.observe(main || document.body, { childList: true, subtree: true });
              // wait briefly for possible late-added images
              await new Promise(res => setTimeout(res, 800));
              mo.disconnect();
              const newPending = addedImgs.filter(i => !i.complete);
              if(newPending.length){
                await Promise.race([
                  Promise.all(newPending.map(i => new Promise(res => { i.addEventListener('load', res); i.addEventListener('error', res); }))),
                  new Promise(res => setTimeout(res, 2000))
                ]);
              }
            }catch(e){ console.warn('wait for images failed', e); }

            // retry a few times to handle other late layout shifts (fonts, 3rd-party widgets)
            for(let i=0;i<8;i++){
              await new Promise(r=>setTimeout(r, 160));
              doScroll();
              const pos = useMain && main ? main.scrollTop : (window.scrollY || document.documentElement.scrollTop || 0);
              console.log('restore: attempt', i+1, 'measured', pos, 'target', target, 'useMain', useMain);
              if(Math.abs(pos - target) < 8) break;
            }
            // As a final attempt, if a component signals readiness via window.sectionReady, respect it
            if(typeof window.__lastSectionReady === 'number'){
              const pos = main ? main.scrollTop : (window.scrollY || window.pageYOffset || 0);
              console.log('restore: sectionReady timestamp', window.__lastSectionReady, 'measured', pos);
            }
          })(pendingScroll, preferMain);
          // Clear stored values so they aren't reused
          sessionStorage.removeItem('lastSection');
          sessionStorage.removeItem('lastScroll');
          sessionStorage.removeItem('lastScrollMain');
          sessionStorage.removeItem('lastScrollWindow');
            // Drain any queued load requests (skip any 'home' requests)
            const q = window.__pendingLoadQueue || [];
            window.__pendingLoadQueue = [];
            (async function drain(){
              for(const req of q){
                // Skip any queued home requests so restore isn't overwritten
                if(req && req.section === 'home'){
                  console.log('drain: skipping queued home request');
                  try{ req.resolve(); }catch(e){}
                  continue;
                }
                try{
                  await window.__performLoadSection(req.section);
                  req.resolve();
                }catch(err){ req.reject(err); }
              }
              // NOW unset the restoring flag after all queued items are done
              console.log('drain complete: clearing __restoringSection flag');
              window.__restoringSection = false;
              // Signal to app.js that restore just completed so it skips default load
              window.__restoreJustCompleted = true;
            })();
        }).catch((err)=>{
          console.error('Error loading section for restore', err);
          // ensure we clear the restoring flag so normal navigation resumes
          sessionStorage.removeItem('lastSection');
          sessionStorage.removeItem('lastScroll');
          window.__restoringSection = false;
        });
      } else {
        console.log('No pending section detected, loading home via else block');
        window.loadSection('home');
      }
    })();
  </script>
